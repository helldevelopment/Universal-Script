AimButton.MouseButton1Click:Connect(function()
	if _G.aimbot == false then
		AimbotOnOrOff.Text = "On"
		AimSliderCircle:TweenPosition(UDim2.new(0.5, 0,0.122, 0),"In","Linear",0.05,true)
		AimSliderCircle:TweenSize(UDim2.new(0.566, 0,0.737, 0),"In","Linear",0.05,true)
		wait(0.05)
		AimSliderCircle:TweenSize(UDim2.new(0.301, 0,0.737, 0),"In","Linear",0.05,true)
		AimSliderCircle:TweenPosition(UDim2.new(0.8, 0,0.122, 0),"In","Linear",0.05,true)
		_G.aimbot = true
		function getClosest()
			local closestdistance = math.huge
			local target = nil

			for i,v in pairs(game.Players:GetChildren()) do
				if v ~= game.Players.LocalPlayer  then
					if v.Character and v.Character:FindFirstChild("Head") and v.Character.Humanoid.Health > 0 and _G.aimbot then
						if TeamCheck == true then
							if game.Players.LocalPlayer.Team ~= v.Team then
								local distance = (game.Players.LocalPlayer.Character.HumanoidRootPart.Position - v.Character.HumanoidRootPart.Position).magnitude
								if distance < closestdistance then
									closestdistance = distance
									target = v
								end
							end
						else
							local distance = (game.Players.LocalPlayer.Character.HumanoidRootPart.Position - v.Character.HumanoidRootPart.Position).magnitude
							if distance < closestdistance then
								closestdistance = distance
								target = v
							end
						end
					end
				end
			end
			return target
		end
	else
		AimbotOnOrOff.Text = "Off"
		AimSliderCircle:TweenPosition(UDim2.new(0.5, 0,0.122, 0),"In","Linear",0.05,true)
		AimSliderCircle:TweenSize(UDim2.new(0.566, 0,0.737, 0),"In","Linear",0.05,true)
		wait(0.05)
		AimSliderCircle:TweenSize(UDim2.new(0.301, 0,0.737, 0),"In","Linear",0.05,true)
		AimSliderCircle:TweenPosition(UDim2.new(0.2, 0,0.122, 0),"In","Linear",0.05,true)
		_G.aimbot = false

	end
end)
